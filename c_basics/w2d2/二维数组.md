## 选择排序


int main()
{
    int a[5] = {1, 2, 3, 4, 5};
    for (int i = 0; i < len; i++)
    {
        int m = i;
        for (int j = 0; j < len - 1 - i; j++)
        {
            if(a[m] < a[j])
            {
                m = j;
            }
        }
        if (i != m)
        {
            a[m] = a[i] ^ a[m];
            a[i] = a[i] ^ a[m];
            a[m] = a[i] ^ a[m];
        }
    }
    for(int i = 0; i < len; i++)
    {
        printf("%d\n", a[i]);
    }
}
